  20: let create x d l r =
  21:   let hl = height l in
  22:   let hr = height r in

  25: let bal x d l r =
  26:   let hl = height l in
  27:   let hr = height r in

  29:       match r with
  30:           Empty -> assert (0 = 1); assert false (* invalid_arg «Map.bal» *)
  31:         | Node(rv, rd, rl, rr, h) ->
  32:             if height rr >= height rl then
  33:               create rv rd (create x d l rl) rr
  34:             else begin
  35:               match rl with
  36:                   Empty -> assert (0 = 1); assert false (* invalid_arg «Map.bal» *)
  37:                 | Node(rlv, rld, rll, rlr, h) ->
  38:                     create rlv rld (create x d l rll) (create rv rd rlr rr)
  39:             end
  40:     else if hl > hr + 2 then
  41:       match l with
  42:           Empty -> assert (0 = 1); assert false (* invalid_arg «Map.bal» *)
  43:         | Node (lv, ld, ll, lr, h) ->
  44:             if height ll >= height lr then
  45:               create lv ld ll (create x d lr r)

  50:                     create lrv lrd (create lv ld ll lrl) (create x d lrr r)

  56: let rec add x data t =
  57:   match t with
  58:       Empty ->
  59:         Node(x, data, Empty, Empty, 1)
  60:     | Node(v, d, l, r, h) ->
  61:         (* let c = Ord.compare x v in *)
  62:         if x = v (* c = 0 *) then
  63:           Node(x, data, l, r, h) 
  64:         else if x < v (* c < 0 *) then
  65:           bal v d (add x data l) r
  66:         else
  67:           bal v d l (add x data r)
