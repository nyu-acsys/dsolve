   5: type 'a t =
   6:     Empty
   7:       (* meaning: node of left el, left n.els, element, right el, right n. els, heigth *)
   8:   | Node of 'a t * int * 'a * 'a t * int * int
   9: 
  10: let empty = Empty
  11: 
  12: let height t =
  13:   match t with
  14:       Empty -> 0
  15:     | Node(_, _, _, _, _, h) -> h
  16: 
  17: let length t =
  18:   match t with
  19:       Empty -> 0
  20:     | Node (_, cl, _, _, cr, _) -> 1 + cl + cr
  21:         
  22: let makenode l d r =
  23:   let (hl, cl) = match l with
  24:       Empty -> (0,0)
  25:     | Node(_,lcl,_,_,lcr,h) -> (h, lcl + lcr + 1) in
  26:   let (hr, cr) = match r with
  27:       Empty -> (0,0)
  28:     | Node(_,rcl,_,_,rcr,h) -> (h, rcl + rcr + 1) in
  29:     Node(l, cl, d, r, cr, (if hl >= hr then hl + 1 else hr + 1))
  30: 
  31: let rec create d n =
  32:   if n = 0 then Empty else
  33:     let ml = n / 2 in 
  34:     let mr = n - ml - 1 in 
  35:     let l = create d ml in
  36:     let r = create d mr in (* defer this particular property to runtime *)
  37:       if height l >= height r + 3 or height l <= height r - 3 then
  38:         assert false 
  39:       else
  40:         makenode l d r
  41: 
  42: (* bal assumes that l and r are of similar height *)
  43: let bal l d r =
  44:   let hl = 
  45:     match l with 
  46:         Empty -> 0
  47:       | Node(_,_,_,_,_,h) -> h in
  48:   let hr = 
  49:     match r with 
  50:         Empty -> 0 
  51:       | Node(_,_,_,_,_,h) -> h in
  52:     if hl > hr + 2 then begin
  53:       match l with
  54:           Empty -> assert (1 = 0); assert false (*invalid_arg «Vec.bal»*)
  55:         | Node(ll, lll, ld, lr, llr, h) ->
  56:             if height ll >= height lr then
  57:               makenode ll ld (makenode lr d r)
  58:             else begin
  59:               match lr with
  60:                   Empty -> assert (1 = 0); assert false (*invalid_arg «Vec.bal»*)
  61:                 | Node(lrl, llrl, lrd, lrr, llrr, h) ->
  62:                     makenode (makenode ll ld lrl) lrd (makenode lrr d r)
  63:             end
  64:     end else if hr > hl + 2 then begin
  65:       match r with
  66:           Empty -> assert (1 = 0); assert false (*invalid_arg «Vec.bal»*)
  67:         | Node(rl, lrl, rd, rr, lrr, h) ->
  68:             if height rr >= height rl then
  69:               makenode (makenode l d rl) rd rr
  70:             else begin
  71:               match rl with
  72:                   Empty -> assert (1 = 0); assert false (*invalid_arg «Vec.bal»*)
  73:                 | Node(rll, lrll, rld, rlr, lrlr, h) ->
  74:                     makenode (makenode l d rll) rld (makenode rlr rd rr)
  75:             end
  76:     end 
  77:     else makenode l d r

  79: (* recbal as originally included in vec.ml did not provide a balancing guarantee. 
  80:  * this has been fixed below by adding at worst a small constant number of rotations per level.
  81:  * the fixed version also typechecks. *)
  82:       
  83: (* This is a recursive version of balance, which balances a tree all the way down. 
  84:    The trees l and r can be of any height, but they need to be internally balanced.  
  85:    Useful to implement concat. *)
  86: let rec recbal l d r =
  87:   let hl = match l with Empty -> 0 | Node(_,_,_,_,_,h) -> h in
  88:   let hr = match r with Empty -> 0 | Node(_,_,_,_,_,h) -> h in
  89:     if hl > hr + 2 then begin
  90:       match l with
  91:           Empty -> assert (1 = 0); assert false (*invalid_arg «Vec.bal»*)
  92:         | Node(ll, _, ld, lr, _, h) ->
  93:             if height ll >= height lr then
  94:               bal ll ld (recbal lr d r)
  95:             else begin
  96:               match lr with
  97:                   Empty -> assert (1 = 0); assert false (*invalid_arg «Vec.bal»*)
  98:                 | Node(lrl, _, lrd, lrr, _, h) ->
  99:                     let nr = recbal lrr d r in
 100:                       if height nr <= height lr - 3 then
 101:                         makenode ll ld (bal lrl lrd nr)
 102:                       else
 103:                         makenode (makenode ll ld lrl) lrd nr
 104:             end
 105:     end else if hr > hl + 2 then begin
 106:       match r with
 107:           Empty -> assert (1 = 0); assert false (*invalid_arg «Vec.bal»*)
 108:         | Node(rl, _, rd, rr, _, h) ->
 109:             if height rr >= height rl then
 110:               bal (recbal l d rl) rd rr
 111:             else begin
 112:               match rl with
 113:                   Empty -> assert (1 = 0); assert false (*invalid_arg «Vec.bal»*)
 114:                 | Node(rll, _, rld, rlr, _, h) ->
 115:                     let nl = recbal l d rll in
 116:                       if height nl <= height rl - 3 then
 117:                         makenode (bal nl rld rlr) rd rr
 118:                       else
 119:                         makenode nl rld (makenode rlr rd rr)
 120:             end
 121:     end 
 122:     else makenode l d r

 124: let is_empty t = 
 125:   match t with
 126:     | Empty -> true
 127:     | Node(_, _, _, _, _, _) -> false

 129: let singleton d = Node (Empty, 0, d, Empty, 0, 1)

 131: let rec get i t =
 132:   match t with
 133:       Empty -> assert (1 = 0); assert false (*raise Vec_index_out_of_bounds*)
 134:     | Node (l, cl, d, r, cr, _) -> 
 135:         if i < cl then get i l 
 136:         else if i > cl then get (i - cl - 1) r 
 137:         else d 
 138:           
 139: let rec set i d t =
 140:   match t with
 141:       Empty -> assert (1 = 0); assert false (*raise Vec_index_out_of_bounds*)
 142:     | Node (l, cl, dd, r, cr, _) -> 
 143:         if i < cl then makenode (set i d l) dd r
 144:         else if i > cl then makenode l dd (set (i - cl - 1) d r)
 145:         else makenode l d r
 146: 
 147: let rec append d t =
 148:   match t with
 149:       Empty -> Node (Empty, 0, d, Empty, 0, 1)
 150:     | Node (l, ll, dd, r, lr, h) -> 
 151:         bal l dd (append d r)
 152:           
 153: (* although it's not obvious from setappend, length v > 0 due to dep on set *)
 154: let setappend d0 d i v =
 155:   let l = length v in 
 156:     if l > i then set i d v 
 157:     else begin
 158:       let rec app_rec n v =
 159:         if n = (l-1) then v else append d0 (app_rec (n-1) v) in 
 160:         append d (app_rec (i-1) v)
 161:     end

 163: let rec leftmost t =
 164:   match t with
 165:       Empty -> assert (1 = 0); assert false (*raise Vec_index_out_of_bounds*)
 166:     | Node(l, ll', d, r, lr', h) -> 
 167:         match l with
 168:           | Empty -> d
 169:           | Node(ll, lll, d', lr, llr, h') -> leftmost l
 170:               
 171: let rec remove_leftmost t =
 172:   match t with
 173:       Empty -> assert (1 = 0); assert false (*invalid_arg «Vec.remove_min_elt»*)
 174:     | Node(l, ll, d, r, lr, h) ->
 175:         match l with
 176:           | Empty -> r
 177:           | Node(ll, lll, ld, lr, llr, h') -> bal (remove_leftmost l) d r
 178:               
 179: (* merge uses bal, not recbal, so it only works for trees of similar height *)
 180: let merge t1 t2 =
 181:   match t1 with
 182:     | Empty -> t2
 183:     | Node(_, _, _, _, _, _) ->
 184:         match t2 with
 185:           | Empty -> t1
 186:           | Node(_, _, _, _, _, _) ->
 187:               let d = leftmost t2 in
 188:                 bal t1 d (remove_leftmost t2)
 189: 
 190: (* Concat works also for t1 and t2 of very different heights *)
 191: let concat t1 t2 =
 192:   match t1 with
 193:     | Empty -> t2
 194:     | Node(_, _, _, _, _, _) ->
 195:         match t2 with
 196:           | Empty -> t1
 197:           | Node(_, _, _, _, _, _) ->
 198:               let d = leftmost t2 in
 199:                 recbal t1 d (remove_leftmost t2)

 201: let rec pop i t =
 202:   match t with
 203:       Empty -> assert (1 = 0); assert false (*raise Vec_index_out_of_bounds*)
 204:     | Node(l, cl, d, r, cr, h) ->
 205:         if i < cl then 
 206:           let (e, v) = pop i l in 
 207:             (e, bal v d r)
 208:         else if i > cl then 
 209:           let (e, v) = pop (i - cl - 1) r in 
 210:             (e, bal l d v)
 211:         else (d, merge l r)
 212: 
 213: let rec remove i t =
 214:   match t with
 215:       Empty -> assert (1 = 0); assert false (*raise Vec_index_out_of_bounds*)
 216:     | Node(l, cl, d, r, cr, h) ->
 217:         if i < cl then 
 218:           bal (remove i l) d r 
 219:         else if i > cl then 
 220:           bal l d (remove (i - cl - 1) r)
 221:         else merge l r 
 222:           
 223: let rec insert i d t =
 224:   match t with
 225:       Empty -> begin
 226:         if i = 0 
 227:         then Node (Empty, 0, d, Empty, 0, 1)
 228:         else (assert (1 = 0); assert false) (*raise Vec_index_out_of_bounds*)
 229:       end
 230:     | Node(l, cl, dd, r, cr, h) -> 
 231:         if i < cl then bal (insert i d l) dd r 
 232:         else if i > cl then bal l dd (insert (i - cl - 1) d r)
 233:         else bal l d (insert 0 dd r)
 234: 
 235: let rec sub i j t = 
 236:   match t with 
 237:       Empty -> Empty
 238:     | Node (l, cl, dd, r, cr, _) -> 
 239:         if i >= j then Empty
 240:           (* Important for sharing *)
 241:         else if i <= 0 && j >= cl + cr + 1 then t
 242:         else begin 
 243:           if j <= cl then sub i j l 
 244:           else if j = cl + 1 then append dd (sub i cl l)
 245:           else if i = cl then insert 0 dd (sub 0 (j - cl - 1) r)
 246:           else if i > cl then sub (i - cl - 1) (j - cl - 1) r
 247:           else begin
 248:             (* dd straddles the interval *)
 249:             let ll = sub i cl l in 
 250:             let rr = sub 0 (j - cl - 1) r in 
 251:               recbal ll dd rr
 252:           end
 253:         end

 255: let rec iteri t f = 
 256:   let rec offsetiteri t' k =
 257:     match t' with
 258:         Empty -> ()
 259:       | Node(l, cl, d, r, _, _) ->
 260:           offsetiteri l k;
 261:           f (k + cl) d;
 262:           offsetiteri r (k + cl + 1)
 263:   in offsetiteri t 0

 265: let rangeiteri i j t f  = 
 266:   let rec offsetrangeiteri k i' j' t' = 
 267:     match t' with
 268:         Empty -> ()
 269:       | Node(l, cl, d, r, cr, _) ->
 270:           if i' < j' then begin 
 271:             if i' < cl && j' > 0 then offsetrangeiteri k i' j' l else (); 
 272:             if i' <= cl && j' > cl then f (k + cl) d else ();
 273:             if j' > cl + 1 && i' <= cl + cr + 1 then offsetrangeiteri (k + cl + 1) (i' - cl - 1) (j' - cl - 1) r else ()
 274:           end else ()
 275:   in offsetrangeiteri 0 i j t

 345: let rec to_array t = 
 346:   match t with 
 347:       Empty -> [||]
 348:     | Node (l, cl, d, r, cr, _) -> 
 349:         begin 
 350:           (* Creates the array *)
 351:           let n = (cl + cr + 1) in 
 352:           let a = Array.make n d in
 353:             (* and fills it *)
 354:           let rec fill k t' =
 355:             match t' with
 356:                 Empty -> a
 357:               | Node (l, cl, d, r, _, _) -> begin
 358:                   ignore (fill k l); 
 359:                   Array.set a (k + cl) d;
 360:                   fill (k + cl + 1) r
 361:                 end
 362:           in fill 0 t 
 363:         end
