   5: type 'a rlist = 
   6:   | Nil 
   7:   | One of 'a
   8:   | Even of 'a rlist * 'a rlist
   9:   | Odd  of 'a * 'a rlist * 'a rlist
  10: 
  11: let rec sz l = 
  12:   match l with
  13:     | Nil -> 0
  14:     | One x -> 1
  15:     | Even (l1, l2) -> (sz l1) + (sz l2)
  16:     | Odd (_, l1, l2) -> (1 + (sz l1) + (sz l2)) 
  17: 
  18: let rec cons x xs =
  19:   match xs with
  20:     | Nil -> (One x)
  21:     | One y -> (Even(One(x), One(y)))
  22:     | Even(l1, l2) -> (Odd(x, l1, l2))
  23:     | Odd(y, l1, l2) -> (Even(cons x l1, cons y l2))
  24: 
  25: let rec makelist n =
  26:   if n = 0 then Nil else
  27:     let l = (makelist (n-1)) in
  28:       cons n l 
  29: 
  30: let rec uncons xs =
  31:   match xs with
  32:     | Nil -> assert (1 = 0); assert false
  33:     | One x -> (x, Nil)
  34:     | Even(l1, l2) ->
  35:         let (x, l1) = uncons l1 in
  36:         let (y, l2) = uncons l2 in
  37:           begin
  38:             match l1 with
  39:                 Nil -> (x, One y)
  40:               | One (_) -> (x, Odd(y, l1, l2))
  41:               | Even (_, _) -> (x, Odd(y, l1, l2))
  42:               | Odd (_, _, _) -> (x, Odd(y, l1, l2))
  43:           end
  44:     | Odd(x, l1, l2) -> (x, Even(l1, l2))
  45: 
  46: let head_safe l =
  47:   let (x, _) = uncons l in x
  48: 
  49: let tail_safe l =
  50:   let (x, l) = uncons l in l
  51: 
  52: let head l = 
  53:   match l with
  54:     | Nil -> assert (0 = 1); assert false
  55:     | One _ -> head_safe l
  56:     | Even _ ->  head_safe l
  57:     | Odd _ ->  head_safe l
  58: 
  59: let tail l =
  60:   match l with
  61:     | Nil -> assert (0 = 1); assert false
  62:     | One _ -> tail_safe l
  63:     | Even _ ->  tail_safe l
  64:     | Odd _ ->  tail_safe l
  65: 
  66: let rec lookup l i =
  67:   match l with
  68:     | Nil -> assert (1 = 0); assert false
  69:     | One x -> if i = 0 then x else let _ = assert (1 = 0) in assert false 
  70:     | Even (l1, l2) ->
  71:         if 2 * (i / 2) = i then lookup l1 (i / 2) else lookup l2 (i / 2) 
  72:     | Odd(x, l1, l2) ->
  73:         if i = 0 then x
  74:         else if 2 * (i / 2) = i then lookup l2 ((i - 1) / 2) 
  75:         else lookup l1 ((i - 1) / 2)
  76: 
  77: let rec print_rlist l =
  78:   match l with
  79:       Nil -> ()
  80:     | One (_) -> 
  81:         let (x, _) = uncons l in print_int x; print_newline ()
  82:     | Even (_, _) -> 
  83:         let (x, l) = uncons l in print_int x; print_rlist l
  84:     | Odd (_, _, _) ->
  85:         let (x, l) = uncons l in print_int x; print_rlist l
