   5: type 'a dict =
   6:     Empty
   7:   | Black of 'a * 'a dict * 'a dict
   8:   | Red of 'a * 'a dict * 'a dict
   9:   | Purple of 'a * 'a dict * 'a dict
  10: 
  11: let color = function
  12:   | Empty -> 0
  13:   | Black (a, b, c) -> 1
  14:   | Red (a, b, c) -> 2
  15:   | Purple (a, b, c) -> 3
  16: 
  17: let max x y =
  18:   if x > y then x else y
  19: 
  20: let rec height = function
  21:   | Empty -> 0
  22:   | Red (_, l, r) -> (max (height l) (height r))
  23:   | Black (_, l, r) -> (max (height l) (height r)) + 1
  24:   | Purple (_, l, r) -> (max (height l) (height r))
  25: 
  26: let restore_right e lt r =
  27:   match r with
  28:   | Purple (re, rl, rr) ->
  29:       begin match lt with
  30:         | Red (le, ll, lr) ->
  31:             begin match rl with
  32:               | Red _ ->
  33:                   Red (e, Black (le, ll, lr), Black (re, rl, rr))     (* re-color *)
  34:               | Black _   ->
  35:                   begin match rr with
  36:                     |  Red _ ->
  37:                          Red (e, Black (le, ll, lr), Black (re, rl, rr))     (* re-color *)
  38:                     | Black _   -> assert (0 = 1); assert false
  39:                     | Purple _ -> assert (0 = 1); assert false
  40:                   end
  41:               | Purple _ -> assert (0 = 1); assert false
  42:             end
  43:         | Black _ ->
  44:             begin match rl with
  45:               | Red (rle, rll, rlr) ->
  46:                   (* l is black, deep rotate *)
  47:                   Black (rle, Red (e, lt, rll), Red (re, rlr, rr))
  48:               | Black _   ->
  49:                   (* l is black, shallow rotate *)
  50:                   Black(re, Red(e, lt, rl), rr)
  51:               | Purple _ -> assert (0 = 1); assert false
  52:             end
  53:         | Purple _ -> assert (0 = 1); assert false
  54:       end
  55:   | Red _   -> Black (e, lt, r)
  56:   | Black _ -> Black (e, lt, r)
  57: 
  58: let restore_left e l rt =
  59:   match l with
  60:   | Purple (le, ll, lr) ->
  61:       begin match rt with
  62:         | Red (re, rl, rr) ->
  63:             begin match lr with
  64:               | Red _ ->
  65:                   Red (e, Black (le, ll, lr), Black (re, rl, rr))     (* re-color *)
  66:               | Black _   ->
  67:                   begin match ll with
  68:                     | Red _ ->
  69:                         Red (e, Black (le, ll, lr), Black (re, rl, rr))     (* re-color *)
  70:                     | Black _   -> assert (0 = 1); assert false
  71:                     | Purple _ -> assert (0 = 1); assert false
  72:                   end
  73:               | Purple _ -> assert (0 = 1); assert false
  74:             end
  75:         | Black _ ->
  76:             begin match lr with
  77:               | Red (lre, lrl, lrr) ->
  78:                   (* r is black, deep rotate *)
  79:                   Black (lre, Red (le, ll, lrl), Red(e, lrr, rt))
  80:               | Black _   ->
  81:                   (* r is black, shallow rotate *)
  82:                   Black (le, ll, Red (e, lr, rt))
  83:               | Purple _ -> assert (0 = 1); assert false
  84:             end
  85:         | Purple _ -> assert (0 = 1); assert false
  86:       end
  87:   | Red _   -> Black (e, l, rt)
  88:   | Black _ -> Black (e, l, rt)

  90: let rec ins1 key d = match d with
  91:   | Black(key1, left, right) ->
  92:       if key = key1 then Black (key, left, right)
  93:       else if key < key1 then restore_left key1 (ins1 key left) right
  94:       else restore_right key1 left (ins1 key right)
  95:   | Red (key1, left, right) ->
  96:       if key = key1 then Red (key, left, right)
  97:       else if key < key1 then
  98:         let l' = ins1 key left in
  99:           begin match l' with
 100:             | Red _     -> Purple (key1, l', right)
 101:             | Black _   -> Red (key1, l', right)
 102:             | Empty     -> Red (key1, l', right)
 103:             | Purple _ -> assert (0 = 1); assert false
 104:           end
 105:       else
 106:         let r' = ins1 key right in
 107:           begin match r' with
 108:             | Red _     -> Purple (key1, left, r')
 109:             | Black _   -> Red (key1, left, r')
 110:             | Empty     -> Red (key1, left, r')
 111:             | Purple _ -> assert (0 = 1); assert false
 112:           end
 113:   | Empty     -> Red (key, Empty, Empty)
 114:   | Purple _ -> assert (0 = 1); assert false

 116: let insert dict key =
 117:   let dict = ins1 key dict in
 118:     match dict with
 119:       | Purple (d, lt, rt) -> Black (d, lt, rt) (* re-color *)
 120:       | Red (d, lt, rt)     -> Red (d, lt, rt)
 121:       | Black (d, lt, rt)   -> Black (d, lt, rt)
 122:       | Empty               -> Empty
