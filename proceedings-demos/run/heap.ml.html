  16: (* Heaps *)
  17: 
  18: (* Heaps are encoded as complete binary trees, i.e., binary trees
  19:    which are full expect, may be, on the bottom level.
  20:    These trees also enjoy the heap property, namely the value of any node
  21:    is greater or equal than those of its left and right subtrees.
  22: 
  23:    The representation invariant is the following: the number of nodes in
  24:    the left subtree is equal to the number of nodes in the right
  25:    subtree, or exceeds it by exactly once. In the first case, we use
  26:    the constructor [Same] and in the second the constructor [Diff].
  27:    Then it can be proved that [2^(h-1) <= n <= 2^h] when [n] is the
  28:    number of elements and [h] the height of the tree. *)
  29: 
  30: type 'a t =
  31:   | Empty
  32:   | Same of 'a * 'a t * 'a t (* same number of elements on both sides *)
  33:   | Diff of 'a * 'a t * 'a t (* left has [n+1] nodes and right has [n] *)
  34: 
  35: let rec set_of t = match t with
  36:     Empty -> Myaset.empty
  37:   | Same (x, l, r) -> Myaset.cup (Myaset.sng x) (Myaset.cup (set_of l) (set_of r))
  38:   | Diff (x, l, r) -> Myaset.cup (Myaset.sng x) (Myaset.cup (set_of l) (set_of r))
  39: 
  40: let rec sz t = match t with
  41:     Empty -> 0
  42:   | Same (_, l, r) -> 1 + sz l + sz r
  43:   | Diff (_, l, r) -> 1 + sz l + sz r

  45: let max h = match h with
  46:     Empty -> assert false
  47:   | Same (x, _, _) -> x
  48:   | Diff (x, _, _) -> x

  52: let rec add x h = match h with
  53:   | Empty ->
  54:       Same (x, Empty, Empty)
  55:         (* insertion to the left *)
  56:   | Same (y, l, r) ->
  57:       if x > y then Diff (x, add y l, r) else Diff (y, add x l, r)
  58:         (* insertion to the right *)
  59:   | Diff (y, l, r) ->
  60:       if x > y then Same (x, l, add y r) else Same (y, l, add x r)
  61:       
  62: let maximum h = match h with
  63:   | Empty -> let _ = assert (1 = 0) in assert false (* raise EmptyHeap *)
  64:   | Same (x, l, r) -> (x, Same (x, l, r)) | Diff (x, l, r) -> (x, Diff (x, l, r))

  66: (* extracts one element on the bottom level of the tree, while
  67:    maintaining the representation invariant *)
  68: let rec extract_last h = match h with
  69:   | Empty -> let _ = assert (1 = 0) in assert false (* raise EmptyHeap *)
  70:   | Same (x, l, r) ->
  71:       (match l with
  72:       | Empty ->
  73:           (match r with
  74:           | Empty -> (x, Empty)
  75:           | Same (_, _, _) -> let _ = assert (1 = 0) in assert false
  76:           | Diff (_, _, _) -> let _ = assert (1 = 0) in assert false)
  77:       | Same (_, _, _) ->
  78:           let y,r' = extract_last r in y, Diff (x, l, r')
  79:       | Diff (_, _, _) ->
  80:           let y,r' = extract_last r  in y, Diff (x, l, r'))
  81:   | Diff (x, l, r) -> let y,l' = extract_last l in y, Same (x, l', r)

  83: let ck_assume_empty h = match h with
  84:   | Empty -> h
  85:   | Same(_, _, _) -> let _ = assert (1=0) in assert false
  86:   | Diff(_, _, _) -> let _ = assert (1=0) in assert false

  88: (* returns the topmost element for set invariant *)
  89: (* removes the topmost element of the tree and inserts a new element [x] *)
  90: let rec descent x h = match h with
  91:   | Empty ->
  92:       let _ = assert (1 = 0) in assert false
  93:   | Diff (y, l, r) ->
  94:       (match r with
  95:       | Empty ->
  96:           (match l with
  97:           | Same (z, ll', lr') ->
  98:               let ll = ck_assume_empty ll' in let _ = assert (ll = ll') in
  99:               let lr = ck_assume_empty lr' in let _ = assert (lr = lr') in
 100:               if x > z then (y, Diff (x, Same (z, ll, lr), Empty))
 101:               else (y, Diff (z, Same (x, Empty, Empty), Empty))
 102:           | Diff (_, _, _) -> let _ = assert (1=0) in assert false
 103:           | Empty -> let _ = assert (1=0) in assert false)
 104:       | Diff (_, _, _) ->  (* same code (and proof) for Same and Diff. the only purpose of matching a pattern is to assert !Empty *)
 105:           let (ml, l) = maximum l in
 106:           let (mr, r) = maximum r in
 107:           if x > ml && x > mr then
 108:             (y, Diff (x, l, r))

 110:             let (z, l') = descent x l in
 111:               (y, Diff (ml, l', r))

 113:           let (z, r') = descent x r in
 114:             (y, Diff (mr, l, r'))

 122:               (y, Diff (ml, l', r))

 158:   | Same (x, l, r) ->

 169: let rec iter f = function
 170:   | Empty -> ()
 171:   | Same (x, l, r) -> iter f l; f x; iter f r
 172:   | Diff (x, l, r) -> iter f l; f x; iter f r
